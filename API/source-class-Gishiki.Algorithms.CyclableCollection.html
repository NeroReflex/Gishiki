 20: /**
 21:  * This is a generic implementation of a class that is a container for something
 22:  * else, and what is stored inside can be accessed either as an array or with
 23:  * a foreach cycle
 24:  *
 25:  * @author Benato Denis <benato.denis96@gmail.com>
 26:  */
 27: class CyclableCollection implements \Iterator
 28: {
 29:     
 30:     /**
 31:      * All functions of this base class point to this array
 32:      * 
 33:      * @var array the array all punctions are pointing to
 34:      */
 35:     protected $array = array();
 36:     
 37:     /**
 38:      * Setup the new collection with provided key=>value pairs
 39:      * 
 40:      * @param array $setup_values the array containing properties and values
 41:      */
 42:     public function __construct($setup_values = [])
 43:     {
 44:         if (gettype($setup_values) == "array") {
 45:             //type is very important here!
 46:    $this->array = $setup_values;
 47:         }
 48:     }
 49:     
 50:     /**
 51:      * Start up a new iteration cycle: reset the currently active collection element
 52:      */
 53:     public function rewind()
 54:     {
 55:         reset($this->array);
 56:     }
 57:         
 58:     /**
 59:      * Get the currently active collection element in an iteration cycle
 60:      * 
 61:      * Return FALSE if the array is empty
 62:      * 
 63:      * @return mixed the currently active array element
 64:      */
 65:     public function current()
 66:     {
 67:         return current($this->array);
 68:     }
 69:         
 70:     /**
 71:      * Get the index of the currently active collection element
 72:      * 
 73:      * @return int the index of the currently active array element
 74:      */
 75:     public function key()
 76:     {
 77:         return key($this->array);
 78:     }
 79:         
 80:     /**
 81:      * Get the next collection element after the currently active one and switch
 82:      * the currently active collection element with the following one
 83:      * 
 84:      * Return FALSE if the array is empty
 85:      * 
 86:      * @return mixed the next array element
 87:      */
 88:     public function next()
 89:     {
 90:         return next($this->array);
 91:     }
 92:         
 93:     /**
 94:      * Get an element of the collection as it would be an object property
 95:      * 
 96:      * Return NULL if the array doesn't contain the given key
 97:      * 
 98:      * @param  int|string $key the index of the array element to be accessed
 99:      * @return mixed      the requested array element or NULL
100:      */
101:     public function &__get($key)
102:     {
103:         //return the chosen field
104:         if ($this->exists_key($key)) {
105:             return $this->array[$key];
106:         } else {
107:             return null;
108:         };
109:     }
110:     
111:     /**
112:      * Set an element of the collection as it would be an object property
113:      * 
114:      * @param string $key   the of the property to be modified
115:      * @param mixed  $value the value to be assigned to the property
116:      */
117:     public function __set($key, $value)
118:     {
119:         $this->array[$key] = $value;
120:     }
121:         
122:     /**
123:      * Check if the currently active collection element is a valid and existing one
124:      * 
125:      * @return bool TRUE if the currently accessed field is a valid one
126:      */
127:     public function valid()
128:     {
129:         $key = key($this->array);
130:         return ((array_key_exists($key, $this->array)) && ($key !== null && $key !== false));
131:     }
132:     
133:     /**
134:      * Get the previous field preceding the currently active one and switch
135:      * the currently active field with the preceding one
136:      * 
137:      * Return FALSE if the array is empty
138:      * 
139:      * @return mixed the previous field
140:      */
141:     public function previous()
142:     {
143:         return prev($this->array);
144:     }
145:     
146:     /**
147:      * Get the last element of the array and change the currently active array 
148:      * element with the last one.
149:      * 
150:      * Return FALSE if the array is empty
151:      * 
152:      * @return mixed the last element of the array
153:      */
154:     public function end()
155:     {
156:         return end($this->array);
157:     }
158:     
159:     /**
160:      * Perform a search withing the current collection to find the given value.
161:      * 
162:      * Return FALSE if the array doesn't contains the searched value
163:      * 
164:      * @param  mixed           $value the value to be searched
165:      * @return int|string|bool the index of the searched value, or FALSE
166:      */
167:     public function search($value)
168:     {
169:         return array_search($value, $this->array);
170:     }
171:     
172:     /**
173:      * Check if the given key is part of the current collection
174:      * 
175:      * @param  int|string $key the key to be checked
176:      * @return bool       TRUE if the given key exists
177:      */
178:     public function exists_key($key)
179:     {
180:         return array_key_exists($key, $this->array);
181:     }
182:     
183:     /**
184:      * Insert the given element into the collection
185:      * 
186:      * @param mixed $value the value to insert
187:      */
188:     public function emplace($value)
189:     {
190:         $this->array[] = $value;
191:     }
192:     
193:     /**
194:      * Get the number of entries in the current data collection
195:      * 
196:      * @return int the number of elements inside the current collection
197:      */
198:     public function count()
199:     {
200:         return count($this->array);
201:     }
202:     
203:     /**
204:      * Check weather the current collection is empty (has no elements)
205:      * 
206:      * @return bool TRUE only if the current collection is empty
207:      */
208:     public function is_empty()
209:     {
210:         return ($this->count() <= 0);
211:     }
212:     
213:     /**
214:      * Get the currently managed collectionin form as a native array
215:      * 
216:      * @return array the current collection
217:      */
218:     public function __invoke()
219:     {
220:         return $this->array;
221:     }
222: }
